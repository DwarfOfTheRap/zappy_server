# cmake minimal version
cmake_minimum_required(VERSION 2.6)

# add testing
enable_testing()

# project declaration
project(TestZappyServer)

if (APPLE AND EXISTS /goinfre)

    $ENV{HOME}/.brew/include
    ${CMAKE_CURRENT_SOURCE_DIR}/../src

    ${CMAKE_CURRENT_SOURCE_DIR}/../src
